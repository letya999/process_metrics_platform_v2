"SELECT
    fv.full_date AS report_date,
    SUM(fv.value) AS remaining_sp
FROM metrics.v_facts AS fv
LEFT JOIN clean_jira.sprints AS s
  ON fv.entity_id = s.id::text
  AND fv.entity_type = 'sprint'
WHERE fv.calc_code = 'sprint_burndown_remaining_sp'

  -- 1. Фильтр по проекту
  [[ AND {{project_key}} ]]

  -- 2. Фильтр по названию спринта
  [[ AND {{sprint_name}} ]]

  -- 3. Фильтр по дате старта спринта
  [[ AND {{sprint_start_date}} ]]

  -- 4. Срез по типу задачи (Теперь работает как полноценный Dropdown!)
  -- ТРЮК METABASE:
  -- Если пользователь выберет тип (например 'Bug'), строка превратится в: AND fv.slice_value = 'Bug' -- AND fv.slice_rule_id IS NULL
  -- Если ничего не выбрано, строка исчезнет, и останется: AND fv.slice_rule_id IS NULL (базовый расчет без задвоений)
  [[ AND {{issue_type}} -- ]] AND fv.slice_rule_id IS NULL

GROUP BY fv.full_date
ORDER BY fv.full_date ASC;"
